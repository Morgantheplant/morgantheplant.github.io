var makeLinkedList = function(){
  var list = {};
  list.head = null;
  list.tail = null;

  list.addToTail = function(value){
    var node = makeNode(value);
    if(list.head === null) {
      list.head = node;
    }
    if (list.tail !== null) {
      list.tail.next = node;
      node.previous = list.tail;
    }
    list.tail = node;
  };



  list.addToHead = function(value){
    var node = makeNode(value);
    if(list.tail === null) {
      list.tail = node;
    }
    if (list.head === null) {
      list.head = node;
    }
    list.head.previous = node;
    node.next = list.head;
    list.head = node;
  };

  list.removeHead = function(){
    var header = list.head.value;
    list.head = list.head.next;
    return  header;
  };

  list.removeTail = function() {
    var tailValue = list.tail.value;
    list.tail = list.tail.previous;
    return tailValue;
  };

  list.contains = function(target){
    var contains = false;
    var search = function(node){
      if (node.value === target){
        contains = true;
      }

      if(node !== list.tail) {
        search(node.next);
      }

    };
    search(list.head);
    return contains;
  };
    return list;
};


var makeNode = function(value) {
  var node = {};

  node.value = value;
  node.next = null;
  node.previous = null;
  return node;
};
